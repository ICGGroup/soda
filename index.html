Selenium Client For NodeJS

npm install soda
var soda = require('soda');

var browser = soda.createSauceClient({
    'url': 'http://yourwebsite/'
  , 'username': ''
  , 'access-key': ''
  , 'os': 'Linux'
  , 'browser': 'firefox'
  , 'browser-version': '3.'
  , 'max-duration': 300
});

browser
  .chain
  .session()
  .open('/')
  .clickAndWait('//input[@value="Submit"]')
  .assertTextPresent('Welcome')
  .clickAndWait('link=Log out')
  .testComplete()
  .end(function(err){
    if (err) throw err;
    console.log('done');
  });
  • Soda is a Selenium RC client for Node.js, enabling developers to write automated acceptance tests with an expressive API.
  • Soda has built-in SauceLabs integration, bringing automated acceptance testing to the cloud with Sauce OnDemand.
